fn process_keypress(&mut self) -> Result<(), std::io::Error> {
    let pressed_key = read_key()?;

    if let Key::Ctrl('q') | Key::Ctrl('x') = pressed_key {
        self.should_quit = true;
    }

    // 和下面这句等价：
    match pressed_key {
