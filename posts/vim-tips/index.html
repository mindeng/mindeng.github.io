# [d]elete all lines not(!) matching patterns
:g!/pattern/d
:v/pattern/d

# 匹配多个单词
:v/onStart\|onStop/d

# 忽略大小写，可以直接 `:set ignorecase`，或者：
# 强制忽略大小写：\c
:g/\cpattern/z#.1|echo "================================"
# 强制匹配大小写：\C
:g/\Cpattern/z#.1|echo "================================"

# 更多信息可查看帮助 `:help /ignorecase`

# 将匹配的行移动到最后吗
:g/pattern/m0
# 将匹配的行移动到最前面（顺序会变成倒序）
:g/pat/m$

# 展示匹配该正则表达式的列表
:g/regular-expression/p
